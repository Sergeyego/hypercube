<q>
select  i.nam, t.nam, e.marka, cast(r.diam as varchar(3)) as dm, r.n_s,
substr(cast(n.dat as char(32)),1,4) as yr,
substr(cast(n.dat as char(32)),1,7) as mn,
substr(cast(n.dat as char(32)),1,10) as dy,
(p.kvo*i.koef)/1000.0 as kvo
from istoch i 
inner join prod_nakl n on i.id=n.id_ist 
inner join prod p on p.id_nakl=n.id 
inner join parti r on r.id=p.id_part 
inner join elrtr e on e.id=r.id_el 
inner join el_types t on t.id=e.id_vid 
where n.dat between :d1 and :d2 and i.id=1 order by e.marka, r.diam
